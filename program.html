Sample Program in Key

Now we have learned about all of the features of Key, we can see an implementation of a program. 

                    
    Main() {
        // Print a greeting
        print: "Welcome to Key!"
    
        // Declare and initialize variables
        int sum = 0
        int numbers = [5, 10, 15, 20, 25]
        int count = 0
    
        // Calculate the sum of the array using a for loop
        for int i till numbers.end() {
            sum += numbers[i]
        }
    
        // Print the result
        print: "The sum of the array is 'sum'." // The sum of the array is 75.
    
        // Check if the sum exceeds a specific value
        if sum > 50 {
            print: "The sum is greater than 50." // The sum is greater than 50.
        } else {
            print: "The sum is 50 or less."
        }
    
        // Call a function to double the sum
        sum = doubleValue(sum)
        print: "After doubling, the sum is 'sum'." // After doubling, the sum is 100."
    }
    
    // Function to double a value
    int fun doubleValue(int value) {
        return value * 2
    }
